# Mobile Portrait
# Optimized layout for 9:16 mobile screens
# Demonstrates portrait-specific composition
#
# Before running: haeccstable> open_monitor monitor2
# Then run: haeccstable> run mobile_portrait.txt

in_var camera = webcam;
out_var mobile = monitor2;

# Full-height portrait layer
layer_obj portrait;
portrait.canvas = (1080, 1920);
camera.cast(portrait);
portrait.transform(0, 0);
portrait.scale(1.0, 1.0);
portrait.rot(0, 0);
portrait.opacity(100);
